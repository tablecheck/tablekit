# Badge

Badge component displays numeric values. To display badge-like text use `BadgeContainer` instead.

<Meta
  title="Components/Badge"
  component={Badge}
  parameters={{
    options: { isToolshown: true },
    controls: {
      exclude: ['children', 'data-testid'],
      sort: 'alpha'
    }
  }}
/>

export const NumericTemplate = (args) => {
  return <Badge {...args} />;
};

export const TextTemplate = (args) => {
  return <BadgeContainer {...args} />;
};
