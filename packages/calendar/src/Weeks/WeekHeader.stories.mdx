The `WeekHeader` export of the date picker supports rendering the titles of one week of dates, it expects in a single array from the `calendar` array.

export const displayDate = new Date('2021-06-21');

export const Template = () => {
  const { calendars } = useCalendarContext();
  return <WeekHeader calendar={calendars[0]} />;
};
